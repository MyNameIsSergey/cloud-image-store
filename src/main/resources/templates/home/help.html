Account

Register:

URL: http://{host}/api/account/register; Method: POST
Args:
String login
String name
String password
String confirmPassword

Login:

URL: http://{host}/api/account/login; Method: POST
Args:
String username
String password
Images

Search:

URL: http://{host}/api/images/search; Method: GET
Args:
String name
Result: Image

Last N images:

URL: http://{host}/api/images/last; Method: GET
Args:
int count
Result: Image[]

Get by id:

URL: http://{host}/api/images/{id}; Method: GET; Authorize: required
Result: Image

Images of user:

URL: http://{host}/api/images/user/{user_id}; Method: GET
Result: Image[]

Images of current user:

URL: http://{host}/api/images/user/; Method: GET
Result: Image[]

Uploading:

URL: http://{host}/api/images/; Method: POST; Authorize: required
Args:
String name
MultipartFile file
Result: Image
Types Image
id: Long
user_id: Long
name: String
url: String
width: Int
height: Int

If method is authorize required

Key: Authorization; Value: Bearer {token}
